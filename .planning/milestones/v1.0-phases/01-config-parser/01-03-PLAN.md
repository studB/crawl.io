Rules for section handling:
  - Headings are H1 nodes (`depth: 1`). Lowercase canonical names expected: `URL`, `Selectors`, `Rules`, `Output`. Be case-insensitive on the heading text when matching (`heading.children[0].value.trim().toLowerCase()`).
  - Required sections: URL, Selectors, Rules. Missing any of them → aggregated error per missing section.
  - Output section is IGNORED (not an error if missing, not an error if present).
  - Unknown H1 sections (e.g., `# Notes`) are ignored silently in v1 (no error, no warning). This mirrors the "prose around YAML is allowed" spirit.
  - A section's content is every node between its heading and the next H1 (or EOF).
  - URL section: first non-empty text line (strip whitespace) wins. If none → error "URL section is empty".
  - Selectors / Rules sections: first fenced code block with `lang` in {'yaml', 'yml'} wins. If none → error "Selectors section has no fenced yaml block" (or equivalent for Rules).

    import { readFile } from 'node:fs/promises';
    import { unified } from 'unified';
    import remarkParse from 'remark-parse';
    import YAML from 'yaml';
    import type { Root, Heading, Code, Text, RootContent } from 'mdast';

    import type { CrawlJob } from './types';
    import { CrawlJobSchema } from './schema';
    import { ConfigParseError } from './errors';

    interface RawSections {
      url?: string;          // raw text line
      selectors?: string;    // raw yaml source
      rules?: string;        // raw yaml source
    }

    const REQUIRED = ['url', 'selectors', 'rules'] as const;
    const KNOWN_SECTIONS = new Set(['url', 'selectors', 'rules', 'output']);

    function splitSections(tree: Root): {
      sections: RawSections;
      issues: string[];
    } {
      const issues: string[] = [];
      const sections: RawSections = {};
      const children = tree.children;

      // Identify indices of H1 headings.
      const h1Indices: number[] = [];
      for (let i = 0; i < children.length; i++) {
        const n = children[i];
        if (n && n.type === 'heading' && (n as Heading).depth === 1) {
          h1Indices.push(i);
        }
      }
      h1Indices.push(children.length); // sentinel

      for (let h = 0; h < h1Indices.length - 1; h++) {
        const start = h1Indices[h]!;
        const end = h1Indices[h + 1]!;
        const heading = children[start] as Heading;
        const firstChild = heading.children[0];
        const rawName =
          firstChild && firstChild.type === 'text'
            ? (firstChild as Text).value.trim().toLowerCase()
            : '';
        if (!KNOWN_SECTIONS.has(rawName)) continue;
        if (rawName === 'output') continue; // Phase 1 ignores Output
        const body = children.slice(start + 1, end);

        if (rawName === 'url') {
          // First paragraph/text with a non-empty trimmed first line wins.
          let url: string | undefined;
          for (const node of body) {
            if (node.type === 'paragraph') {
              const raw = collectText(node).trim();
              if (raw.length > 0) {
                const firstLine = raw.split(/\r?\n/).find((l) => l.trim().length > 0);
                if (firstLine) {
                  url = firstLine.trim();
                  break;
                }
              }
            }
          }
          if (url === undefined) {
            issues.push('URL section is empty (expected a URL on a non-blank line under `# URL`)');
          } else {
            sections.url = url;
          }
        } else if (rawName === 'selectors' || rawName === 'rules') {
          const code = body.find(
            (n): n is Code =>
              n.type === 'code' &&
              (n.lang === 'yaml' || n.lang === 'yml'),
          );
          if (!code) {
            issues.push(
              `${rawName === 'selectors' ? 'Selectors' : 'Rules'} section has no fenced yaml code block`,
            );
          } else {
            if (rawName === 'selectors') sections.selectors = code.value;
            else sections.rules = code.value;
          }
        }
      }

      // Missing-section diagnostics.
      for (const key of REQUIRED) {
        if (sections[key] === undefined && !issues.some((m) => m.toLowerCase().includes(key))) {
          issues.push(`\`# ${key === 'url' ? 'URL' : key.charAt(0).toUpperCase() + key.slice(1)}\` section is missing`);
        }
      }

      return { sections, issues };
    }

    function collectText(node: RootContent | Heading): string {
      const out: string[] = [];
      const visit = (n: any): void => {
        if (!n) return;
        if (typeof n.value === 'string') out.push(n.value);
        if (Array.isArray(n.children)) n.children.forEach(visit);
      };
      visit(node);
      return out.join('\n');
    }

    function tryYamlParse(src: string, label: string): { value: unknown } | { error: string } {
      try {
        return { value: YAML.parse(src) };
      } catch (e) {
        const msg = e instanceof Error ? e.message : String(e);
        return { error: `${label} YAML is invalid: ${msg}` };
      }
    }

    export function parseConfig(
      source: string,
      opts?: { filePath?: string },
    ): CrawlJob {
      const issues: string[] = [];
      const filePath = opts?.filePath;

      // 1. Parse markdown -> AST.
      let tree: Root;
      try {
        const processor = unified().use(remarkParse);
        tree = processor.parse(source) as Root;
      } catch (e) {
        const msg = e instanceof Error ? e.message : String(e);
        throw new ConfigParseError(
          [`Failed to parse markdown: ${msg}`],
          filePath !== undefined ? { filePath } : undefined,
        );
      }

      // 2. Split into sections + collect structural issues.
      const { sections, issues: sectionIssues } = splitSections(tree);
      issues.push(...sectionIssues);

      // 3. YAML-parse the selectors + rules sections (do NOT short-circuit).
      let selectorsRaw: unknown = undefined;
      let rulesRaw: unknown = undefined;

      if (sections.selectors !== undefined) {
        const r = tryYamlParse(sections.selectors, 'Selectors');
        if ('error' in r) issues.push(r.error);
        else selectorsRaw = r.value;
      }
      if (sections.rules !== undefined) {
        const r = tryYamlParse(sections.rules, 'Rules');
        if ('error' in r) issues.push(r.error);
        else rulesRaw = r.value;
      }

      // 4. Schema validation is GATED. It only runs when every structural piece
      //    required by CrawlJobSchema is present:
      //      - sections.url must be a string (URL section parsed successfully),
      //      - selectorsRaw must be defined (Selectors YAML block present AND parsed),
      //      - rulesRaw must be defined (Rules YAML block present AND parsed).
      //    If ANY piece is missing, we already have the structural issue on the
      //    list; re-running safeParse with undefined would produce duplicate
      //    Zod issues for the same root cause (e.g., both "URL section is missing"
      //    and "url: Required"). Skip the schema call entirely in that case.
      const canValidate =
        sections.url !== undefined &&
        selectorsRaw !== undefined &&
        rulesRaw !== undefined;

      if (canValidate) {
        const candidate = {
          url: sections.url,
          selectors: selectorsRaw,
          rules: rulesRaw,
        };
        const result = CrawlJobSchema.safeParse(candidate);
        if (!result.success) {
          for (const issue of result.error.issues) {
            const path = issue.path.length > 0 ? issue.path.join('.') : '<root>';
            issues.push(`${path}: ${issue.message}`);
          }
        } else if (issues.length === 0) {
          return result.data;
        }
      }

      // 5. If we got here, there is at least one problem. Throw aggregated.
      throw new ConfigParseError(
        issues.length > 0 ? issues : ['Unknown config error'],
        filePath !== undefined ? { filePath } : undefined,
      );
    }

    export async function parseConfigFile(path: string): Promise<CrawlJob> {
      let source: string;
      try {
        source = await readFile(path, 'utf8');
      } catch (e) {
        const msg = e instanceof Error ? e.message : String(e);
        throw new ConfigParseError([`Could not read config file: ${msg}`], { filePath: path });
      }
      return parseConfig(source, { filePath: path });
    }
