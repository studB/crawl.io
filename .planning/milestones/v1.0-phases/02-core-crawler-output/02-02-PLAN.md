```ts
export type CrawlErrorCode =
  | 'timeout' | 'selector_miss' | 'network'
  | 'frame_not_found' | 'extraction_failed'
  | 'config_parse' | 'unknown';

export interface CrawlResult {
  status: 'ok' | 'error';
  configPath: string;
  url: string;
  startedAt: Date;
  durationMs: number;
  fields?: Record<string, string>;
  error?: { code: CrawlErrorCode; message: string; stack?: string };
}
```

Phase 1 style reference: the parser intentionally IGNORES any `# Output` section. Its AST walk splits on H1 headings and hands every non-config H1 back as "ignored". That means a file that already has `# Output` still parses cleanly; Phase 2 can append to it without breaking the parser. This plan's job is to make the APPEND idempotent and the RESULT JSON-parseable.

    ```ts
    export function formatTimestamp(date: Date): string;

    export function appendOutput(source: string, entry: string): string;
    export async function writeOutputToFile(configPath: string, entry: string): Promise<void>;
    ```

    Implementation rules:

    - formatTimestamp: use `date.getUTCFullYear()`, `date.getUTCMonth() + 1`, `date.getUTCDate()`, `date.getUTCHours()`, `date.getUTCMinutes()`. Pad each 2-digit field with `.toString().padStart(2, '0')`. Return `${Y}-${M}-${D} ${h}:${m}`.

    - renderEntry(result):
      - Build `const meta = { url: result.url, status: result.status, durationMs: result.durationMs };`
      - If `result.status === 'ok'`:
        - `const fields = result.fields ?? {};`
        - `const payload = { fields, meta };`
        - `const italic = '_count: ' + Object.keys(fields).length + ', duration: ' + result.durationMs + 'ms_';`
      - Else (status === 'error'):
        - Build the error object conditionally so `stack` is included ONLY when present (exactOptionalPropertyTypes compliance — setting `stack: undefined` would violate it AND JSON.stringify would still omit it, but the cleaner approach is a conditional spread):
          ```ts
          const src = result.error ?? { code: 'unknown' as const, message: 'no error detail' };
          const error = {
            code: src.code,
            message: src.message,
            ...(src.stack !== undefined ? { stack: src.stack } : {}),
          };
          const payload = { error, meta };
          const italic = '_error: ' + src.code + ', duration: ' + result.durationMs + 'ms_';
          ```
        - Rationale: the spread `...(src.stack !== undefined ? { stack: src.stack } : {})` adds the `stack` key ONLY when defined. When `stack` is absent, `JSON.stringify` emits no `"stack":` token in the output. When present, it serializes verbatim (newlines become `\n` escapes — still JSON-valid — and Test 9 parses them back).
      - Build the fenced section. To avoid escape-sequence drama with triple backticks in TypeScript source, use a named constant: `const FENCE = '\u0060\u0060\u0060';` then concatenate. Return:
        ```
        '## Run — ' + formatTimestamp(result.startedAt) + '\n\n' +
        italic + '\n\n' +
        FENCE + 'json\n' +
        JSON.stringify(payload, null, 2) + '\n' +
        FENCE + '\n'
        ```
      - The em dash in `## Run —` is the U+2014 character. Type it directly in the source file.

    - appendOutput(source, entry):
      - Normalize: `const src = source.endsWith('\n') ? source : source + '\n';`
      - Detect `# Output` header (case-insensitive) anywhere in the source via `/^# Output\s*$/im.test(src)`.
      - If not found: return `src + '\n# Output\n\n' + entry`.
      - If found: return `src + '\n' + entry` (always append at EOF; do not hunt for the header location).
      - Document this in a top-of-file comment: "appendOutput assumes `# Output` is the last H1 (true for every config this tool produces). New entries are always appended at EOF; the header-detect is solely to avoid creating a duplicate `# Output` heading."

    - writeOutputToFile(configPath, entry):
      - Use `import { readFile, writeFile } from 'node:fs/promises';` at the top of output.ts. Under `module: nodenext` + `type: commonjs`, this compiles to plain `require('node:fs/promises')`.
      - `const source = await readFile(configPath, 'utf8');`
      - `const next = appendOutput(source, entry);`
      - `await writeFile(configPath, next, 'utf8');`
      - Let native fs errors propagate (Test 19). No wrapping.
