       ```typescript
       import { Command } from 'commander';

       /**
        * Build the top-level `crawl` commander program.
        *
        * Pure factory — no side effects; the caller (src/bin/crawl.ts) invokes
        * `.parseAsync(process.argv)` so unit tests can construct and introspect
        * the program without consuming argv or touching process.exit.
        */
       export function buildProgram(): Command {
         const program = new Command();
         program
           .name('crawl')
           .description('Markdown-configured web crawler — one .md file fully describes a crawl job and carries its own results.')
           .version('0.1.0');  // TODO: read from package.json at build time if needed
         registerRunCommand(program);
         program.showHelpAfterError();
         return program;
       }

       /**
        * Entry point used by src/bin/crawl.ts. Parses argv and lets each
        * subcommand's action handler own its own process.exit.
        */
       export async function runCli(argv: readonly string[] = process.argv): Promise<void> {
         const program = buildProgram();
         await program.parseAsync([...argv]);
       }
       ```

| T-04-01 | Information Disclosure | CLI stderr "config not found" message | mitigate | Pass the resolved absolute path through `scrubPaths` before writing to stderr, matching the library's existing path-redaction boundary (inherits Phase 2 MD-04 protection) |
| T-04-02 | Information Disclosure | CLI error summary line | accept+mitigate | runCrawl already runs `scrubPaths` on `error.message` and `error.stack` before returning; CLI just forwards `result.error.message`. No additional scrubbing needed — verified via Task 2 test fixture |
| T-04-03 | Denial of Service | Unbounded argv processing | accept | commander handles argv parsing; `crawl run` takes exactly one positional arg; unknown commands print help and exit. Low-value target, no amplification vector |
