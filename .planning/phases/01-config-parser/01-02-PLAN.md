2. Create src/config/schema.ts exporting exactly three named schemas: `SelectorSpecSchema`, `RulesSchema`, `CrawlJobSchema`. Template (Zod v3 form — adapt to v4 if installed):

    import { z } from 'zod';
    import type { CrawlJob, SelectorSpec } from './types';

    export const SelectorSpecSchema = z
      .object({
        selector: z.string().min(1, 'selector must be a non-empty string'),
        engine: z.enum(['css', 'xpath']).default('css'),
        frame: z.array(z.string().min(1)).optional(),
      })
      .strict()
      .transform((v): SelectorSpec => {
        // Strip frame when undefined so exactOptionalPropertyTypes is satisfied.
        return v.frame === undefined
          ? { selector: v.selector, engine: v.engine }
          : { selector: v.selector, engine: v.engine, frame: v.frame };
      });

    export const RulesSchema = z
      .object({
        waitFor: z.string().min(1).optional(),
        timeout: z.number().int().positive().default(30000),
      })
      .strict()
      .transform((v) => {
        return v.waitFor === undefined
          ? { timeout: v.timeout }
          : { waitFor: v.waitFor, timeout: v.timeout };
      });

    export const CrawlJobSchema = z
      .object({
        url: z.string().url('url must be a valid URL'),
        selectors: z
          .record(z.string().min(1), SelectorSpecSchema)
          .refine((s) => Object.keys(s).length > 0, {
            message: 'selectors must declare at least one named field',
          }),
        rules: RulesSchema,
      })
      .strict();

    // Compile-time guarantee that the schema matches the public type.
    export type _CrawlJobSchemaOutput = z.infer<typeof CrawlJobSchema>;
    const _assertShape: CrawlJob = null as unknown as _CrawlJobSchemaOutput;
    void _assertShape;

Notes:
  - `.strict()` (or z.strictObject in v4) is what powers CFG-06 unknown-key rejection.
  - `.transform()` is used to strip undefined keys so the result satisfies exactOptionalPropertyTypes. Do not skip this — tests below assert `'frame' in parsed === false` when frame is omitted.
  - If the installed Zod is v4 and `z.record(keySchema, valueSchema)` has a different signature, use the v4 equivalent. Preserve behavior.
  - The `_assertShape` trick gives a compile-time check that the inferred schema output assigns to `CrawlJob`. If they drift, tsc fails. Keep it.
